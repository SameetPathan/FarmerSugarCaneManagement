import React, { Component } from 'react';
import { Dropdown, DropdownToggle, DropdownMenu, DropdownItem } from 'reactstrap';

class DocumentDropdown extends Component {
  constructor(props) {
    super(props);
    this.state = {
      dropdownOpen: false,
      selectedDocument: '',
      subDropdownOpen: false
    };

    this.toggle = this.toggle.bind(this);
    this.toggleSubDropdown = this.toggleSubDropdown.bind(this);
    this.handleDocumentSelect = this.handleDocumentSelect.bind(this);
    this.handleAuthorSelect = this.handleAuthorSelect.bind(this);
  }

  toggle() {
    this.setState(prevState => ({
      dropdownOpen: !prevState.dropdownOpen
    }));
  }

  toggleSubDropdown() {
    this.setState(prevState => ({
      subDropdownOpen: !prevState.subDropdownOpen
    }));
  }

  handleDocumentSelect(selectedDoc) {
    this.setState({
      selectedDocument: selectedDoc,
      subDropdownOpen: true
    });
  }

  handleAuthorSelect(selectedAuth) {
    // do something with selected author
  }

  render() {
    return (
      <Dropdown isOpen={this.state.dropdownOpen} toggle={this.toggle}>
        <DropdownToggle caret>
          Select Document
        </DropdownToggle>
        <DropdownMenu right>
          <DropdownItem onClick={() => this.handleDocumentSelect('Doc 1')}>
            Doc 1
            {this.state.selectedDocument === 'Doc 1' && (
              <DropdownMenu right isOpen={this.state.subDropdownOpen} toggle={this.toggleSubDropdown}>
                <DropdownItem onClick={() => this.handleAuthorSelect('Author 1')}>
                  Author 1
                </DropdownItem>
                <DropdownItem onClick={() => this.handleAuthorSelect('Author 2')}>
                  Author 2
                </DropdownItem>
              </DropdownMenu>
            )}
          </DropdownItem>
          <DropdownItem onClick={() => this.handleDocumentSelect('Doc 2')}>
            Doc 2
            {this.state.selectedDocument === 'Doc 2' && (
              <DropdownMenu right isOpen={this.state.subDropdownOpen} toggle={this.toggleSubDropdown}>
                <DropdownItem onClick={() => this.handleAuthorSelect('Author 3')}>
                  Author 3
                </DropdownItem>
                <DropdownItem onClick={() => this.handleAuthorSelect('Author 4')}>
                  Author 4
                </DropdownItem>
              </DropdownMenu>
            )}
          </DropdownItem>
        </DropdownMenu>
      </Dropdown>
    );
  }
}

export default DocumentDropdown;
