import pandas as pd

# Read the Excel file
df = pd.read_excel('filename.xlsx')

# Extract the data from the three columns
column1_data = df['Column1'].tolist()
column2_data = df['Column2'].tolist()
column3_data = df['Column3'].tolist()

# Loop through the rows and print the data for each row
for i in range(len(column1_data)):
    row_data = (column1_data[i], column2_data[i], column3_data[i])
    print(row_data)
