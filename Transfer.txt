import React, { useState } from 'react';

function ImageWithTooltip({ src, title }) {
  const [showTooltip, setShowTooltip] = useState(false);

  const handleButtonClick = () => {
    setShowTooltip(true);
  };

  const handleTooltipClose = () => {
    setShowTooltip(false);
  };

  return (
    <div style={{ position: 'relative' }}>
      <img src={src} alt="" title={title} />
      <button onClick={handleButtonClick}>Show Tooltip</button>
      {showTooltip && (
        <div style={{ position: 'absolute', top: '-30px', left: '50%', transform: 'translateX(-50%)', backgroundColor: '#333', color: '#fff', fontSize: '12px', padding: '5px', borderRadius: '5px' }}>
          {title}
          <button onClick={handleTooltipClose}>Close</button>
        </div>
      )}
    </div>
  );
}

export default ImageWithTooltip;
