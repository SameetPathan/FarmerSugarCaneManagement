import React, { Component } from "react";
import { FormGroup, Label, Input } from "reactstrap";

const docList = [
  {
    "abbreviation": "SPD",
    "docName": "Summary Plan Description in effect during 2022"
  },
  {
    "abbreviation": "AA",
    "docName": "Adoption Agreement, including plan amendments and participation agreements 2022"
  },
  // Add more documents here...
];

class Example extends Component {
  constructor(props) {
    super(props);

    this.state = {
      selectedDoc: null
    };
  }

  handleSelectDoc = (doc) => {
    this.setState({ selectedDoc: doc });
  }

  toggleTab = (tab) => {
    this.setState({ selectedTab: tab });
  }

  render() {
    return (
      <FormGroup>
        <Label for="docSelect">Select a document:</Label>
        <Input type="select" name="docSelect" id="docSelect"
          onChange={(e) => this.handleSelectDoc(e.target.value)}>
          <option value="">Select a document</option>
          {docList.map((doc) => (
            <option value={doc.abbreviation}>{doc.abbreviation}</option>
          ))}
        </Input>
        {this.state.selectedDoc && (
          <FormGroup>
            <Label for="tabSelect">Select tab:</Label>
            <Input type="select" name="tabSelect" id="tabSelect"
              onChange={(e) => this.toggleTab(e.target.value)}>
              <option value="Manual">Manual</option>
              <option value="Bot">Bot</option>
            </Input>
          </FormGroup>
        )}
      </FormGroup>
    );
  }
}

export default Example;
